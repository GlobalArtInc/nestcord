```typescript title="source.autocomplete.ts"
import { Injectable } from '@nestjs/common';
import { AutocompleteInteraction } from 'discord.js';
import { DefaultSources } from 'lavalink-client';
import { AutocompleteInterceptor } from '@globalart/nestcord';

@Injectable()
export class SourceAutocompleteInterceptor extends AutocompleteInterceptor {
    public transformOptions(interaction: AutocompleteInteraction) {
        const focused = interaction.options.getFocused(true);
        let choices: string[];

        if (focused.name === 'source') {
            choices = [DefaultSources.soundcloud] // Обратите внимание, что некоторые источники требуют дополнительных плагинов/конфигурации для надлежащей работы
        }

        return interaction.respond(
        choices
            .filter((choice) => choice.startsWith(focused.value.toString()))
            .map((choice) => ({ name: choice, value: choice })),
        );
    }
}
```

```typescript title="src/app.service.ts"
import { Injectable } from '@nestjs/common';
import { CustomListener, CustomListenerHandler, BaseHandler, ContextOf } from '@globalart/nestcord';
import { User, UserFlagsBitField } from 'discord.js';

export type CustomUserUpdateEvents = {
	userAvatarUpdate: [user: User, oldAvatar: string, newAvatar: string];
};

export type ContextOfCustomUserUpdate<K extends keyof E, E = CustomUserUpdateEvents> = E[K];


@Injectable()
@CustomListener('userUpdate')
export class UserUpdateHandler extends BaseHandler<CustomUserUpdateEvents> {
	@CustomListenerHandler()
	public handleUserAvatarUpdate([oldUser, newUser]: ContextOf<'userUpdate'>) {
		if (oldUser.partial) return;

		if (oldUser.displayAvatarURL() !== newUser.displayAvatarURL()) {
			this.emit(
				'userAvatarUpdate',
				newUser,
				oldUser.displayAvatarURL(),
				newUser.displayAvatarURL()
			);
		}
	}
}
```

```typescript title="app.commands.ts"
import { Injectable, UseInterceptors } from '@nestjs/common';
import { NestCordLavalinkService, PlayerManager } from 'lavalink-client';
import { Context, Options, SlashCommand, SlashCommandContext } from '@globalart/nestcord';
import { QueryDto } from './query.dto';
import { SourceAutocompleteInterceptor } from 'source.autocomplete';

@Injectable()
export class AppCommands {
    public constructor(
        private readonly playerManager: PlayerManager,
        private readonly lavalinkService: NestCordLavalinkService
    ) {}

    @UseInterceptors(SourceAutocompleteInterceptor)
    @SlashCommand({
        name: 'play',
        description: 'воспроизвести трек',
    })
    public async onPlay(
        @Context() [interaction]: SlashCommandContext,
        @Options() { query, source }: QueryDto,
    ) {
        const player =
            this.playerManager.get(interaction.guild.id) ??
            this.playerManager.create({
                ...this.lavalinkService.extractInfoForPlayer(interaction),
                // дополнительные конфигурации:
                selfDeaf: true,
                selfMute: false,
                volume: 100,
            });

        await player.connect();

        const res = await player.search(
            {
                query,
                source: source ?? 'soundcloud'
            },
            interaction.user.id,
        );

        await player.queue.add(res.tracks[0]);
        if (!player.playing) await player.play();

        // Крайне рекомендуется использовать событие `trackStart` для этого объявления
        return interaction.reply({
            content: `Сейчас воспроизводится ${res.tracks[0].info.title}`,
        });
    }
}
```

```python
from telegram import Update
from telegram.ext import ContextTypes
from decorators import TextCommand

@TextCommand("start")
async def start_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Handle /start command"""
    await update.message.reply_text(
        "Welcome! Use /help to see available commands."
    )

@TextCommand("help")
async def help_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Handle /help command"""
    help_text = """
    Available commands:
    /start - Start the bot
    /help - Show this message
    /echo <text> - Echo your message
    """
    await update.message.reply_text(help_text)

@TextCommand("echo")
async def echo_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Handle /echo command"""
    if context.args:
        text = " ".join(context.args)
        await update.message.reply_text(f"Echo: {text}")
    else:
        await update.message.reply_text("Please provide text to echo.")

# Register commands
COMMAND_HANDLERS = {
    "start": start_command,
    "help": help_command,
    "echo": echo_command,
}
```

- Если конфигурация прокси не удастся, приложение будет продолжать работать без прокси
- Параметры прокси влияют на все подключения Discord.js, включая WebSocket и REST API

`ValidationPipe` — это встроенный pipe, который можно использовать для проверки данных, поступающих от клиента. Под капотом он использует пакет [class-validator](https://github.com/typestack/class-validator). `ValidationPipe` предоставляет удобный способ применения правил валидации для всех входящих клиентских полезных нагрузок, где конкретные правила объявляются простыми аннотациями в объявлениях DTO в каждом модуле.

:::caution
Однако при запуске автономного приложения вы не можете использовать глобальные enhancers, поскольку они не зарегистрированы в контексте приложения. Пока [проблема](https://github.com/nestjs/nest/issues/9017) не решена, вы можете использовать локальные enhancers или запустить приложение с помощью метода `app.init()`.
:::

Глобальные команды кэшируются в течение одного часа. Новые глобальные команды будут медленно распространяться по всем серверам и будут гарантированно обновлены только через час. Команды на уровне сервера обновляются мгновенно. Поэтому мы рекомендуем использовать команды на уровне сервера во время разработки и публиковать их как глобальные команды, когда они готовы к использованию.

```typescript title="src/filters/cooldown.filter.ts"
import { ArgumentsHost, Catch, ExceptionFilter } from '@nestjs/common';
import { NestCordArgumentsHost, CooldownException, AnyContext } from '@globalart/nestcord';

@Catch(CooldownException)
export class CooldownExceptionFilter implements ExceptionFilter {
  catch(exception: CooldownException, host: ArgumentsHost) {
    const [interaction] = NestCordArgumentsHost.create(host).getContext<AnyContext>();

    const seconds = (exception.remainingMs / 1000).toFixed(1);
    return interaction.reply({
      content: `⏳ Вы на перезарядке. Попробуйте снова через **${seconds}s**.`,
      ephemeral: true,
    });
  }
}
```

В интересах создания открытой и гостеприимной среды мы, авторы и сопровождающие, обязуемся обеспечить участие в нашем проекте и нашем сообществе свободным от преследований для всех, независимо от возраста, телосложения, инвалидности, этнической принадлежности, гендерной идентичности и самовыражения, уровня опыта, национальности, внешнего вида, расы, религии или сексуальной идентичности и ориентации.

Если коммит отменяет предыдущий коммит, он должен начинаться с `revert:`, за которым следует заголовок отменённого коммита. В теле коммита должно быть написано: `This reverts commit <hash>.`, где hash — это SHA отменяемого коммита.

**Критические изменения** должны начинаться со слова `BREAKING CHANGE:` с пробелом или двумя разрывами строк. Остальная часть сообщения фиксации затем используется для этого.

Лучший способ взаимодействия с пользователями — использование [Slash-команд](https://support.discord.com/hc/en-us/articles/1500000368501-Slash-Commands-FAQ)!
Slash-команды позволяют создавать команды с точными параметрами и вариантами выбора, обеспечивая пользователям наилучший опыт.

Первый шаг — установить nestcord и его зависимость [`Discord.js`](https://discord.js.org)

:::tip
Вам необходимо установить [Node.js](https://nodejs.org/en/) версии 16.0.0 или выше для использования NestCord и Discord.js.

```typescript title="src/discord.module.ts"
import { Module } from '@nestjs/common';
import { AppService } from './app.service';
import { IntentsBitField } from 'discord.js';

@Module({
    imports: [
        NestCordModule.forRoot({
            token: process.env.DISCORD_TOKEN,
            intents: [IntentsBitField.Flags.Guilds],
            development: [process.env.DISCORD_DEVELOPMENT_GUILD_ID]
        })
    ],
    providers: [AppService]
})
export class DiscordModule {}
```

Аргументы модуля являются расширением discord.js [ClientOptions](https://discord.js.org/#/docs/discord.js/stable/typedef/ClientOptions), в дополнение к 3 опциям nestcord: `token`, `prefix` и `development`.

```ts
export interface NestCordModuleOptions extends DiscordClientOptions {
    token: string;
    prefix?: string | (message: Message) => string | Promise<string>;
    development?: Snowflake[] | false;
    skipRegistration?: boolean;
    skipGetCommandInfoFromDiscord?: boolean;
}
```

Если у вас есть команды, использующие свойство `guilds`, глобальный аргумент разработки **не** переопределит его.

Чтобы создать команду с NestCord, вы можете использовать декоратор `SlashCommand`.
