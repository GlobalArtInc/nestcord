```typescript title="anime.interceptor.ts"
import { Injectable } from '@nestjs/common';
import { AutocompleteInteraction } from 'discord.js';
import { AutocompleteInterceptor } from '@globalart/nestcord';

@Injectable()
class AnimeAutocompleteInterceptor extends AutocompleteInterceptor {
    public transformOptions(interaction: AutocompleteInteraction) {
        const focused = interaction.options.getFocused(true);
        let choices: string[];

        if (focused.name === 'anime') {
            choices = ['Hunter x Hunter', 'Naruto', 'One Piece'];
        }

        return interaction.respond(
            choices
                .filter(choice => choice.startsWith(focused.value.toString()))
                .map(choice => ({ name: choice, value: choice }))
        );
    }
}
```

Глобальные команды кэшируются в течение одного часа. Новые глобальные команды будут распространяться медленно на все серверы и будут гарантированно обновлены только через час. Команды сервера обновляются мгновенно. Поэтому мы рекомендуем использовать команды на основе сервера во время разработки и публиковать их в глобальные команды, когда они будут готовы к общественному использованию.

```typescript title="anime-commands.service.ts"
import { Injectable, UseInterceptors } from '@nestjs/common';
import { Context, SlashCommand, Options, SlashCommandContext } from '@globalart/nestcord';
import { AnimeDto } from '/anime.dto';
import { AnimeAutocompleteInterceptor } from './anime.interceptor';

@Injectable()
export class AnimeCommands {
    @UseInterceptors(AnimeAutocompleteInterceptor)
    @SlashCommand({
        name: 'anime',
        description: 'Поиск информации об аниме'
    })
    public async onSearch(@Context() [interaction]: SlashCommandContext, @Options() { anime }: AnimeDto) {
        return interaction.reply({content: `Я нашел аниме ${anime}`});
    }
}
```

**Косые команды** — это новый и захватывающий способ создавать боты и взаимодействовать с ними на **Discord**. С помощью косых команд вам достаточно просто написать `/`, и вы готовы использовать своего любимого бота. Вы легко можете увидеть все команды, которые есть у бота, а проверка валидности и обработка ошибок помогают вам правильно ввести команду с первой попытки.

| Декоратор           | Тип                          | Описание          |
| :------------------ | :---------------------------- | :------------------- |
| `StringOption`      | `string`                      | Опция строки      |
| `NumberOption`      | `number`                      | Опция числа      |
| `IntegerOption`     | `number`                      | Опция целого числа    |
| `BooleanOption`     | `boolean`                     | Опция логического значения     |
| `UserOption`        | `User`                        | Опция пользователя        |
| `MemberOption`      | `GuildMember`                 | Опция участника      |
| `ChannelOption`     | `GuildChannel`                | Опция канала     |
| `RoleOption`        | `Role`                        | Опция роли        |
| `MentionableOption` | `GuildMember` \| `Role` \| `User` | Опция упоминаемого объекта |
| `AttachmentOption`  | `AttachmentOption`            | Опция вложения |
